021/**
022 * Export metrics via the Prometheus Pushgateway.
023 * <p>
024 * The Prometheus Pushgateway exists to allow ephemeral and batch jobs to expose their metrics to Prometheus.
025 * Since these kinds of jobs may not exist long enough to be scraped, they can instead push their metrics
026 * to a Pushgateway. This class allows pushing the contents of a {@link CollectorRegistry} to
027 * a Pushgateway.
028 * <p>
029 * Example usage:
030 * <pre>
031 * {@code
032 *   void executeBatchJob() throws Exception {
033 *     CollectorRegistry registry = new CollectorRegistry();
034 *     Gauge duration = Gauge.build()
035 *         .name("my_batch_job_duration_seconds").help("Duration of my batch job in seconds.").register(registry);
036 *     Gauge.Timer durationTimer = duration.startTimer();
037 *     try {
038 *       // Your code here.
039 *
040 *       // This is only added to the registry after success,
041 *       // so that a previous success in the Pushgateway isn't overwritten on failure.
042 *       Gauge lastSuccess = Gauge.build()
043 *           .name("my_batch_job_last_success").help("Last time my batch job succeeded, in unixtime.").register(registry);
044 *       lastSuccess.setToCurrentTime();
045 *     } finally {
046 *       durationTimer.setDuration();
047 *       PushGateway pg = new PushGateway("127.0.0.1:9091");
048 *       pg.pushAdd(registry, "my_batch_job");
049 *     }
050 *   }
051 * }
052 * </pre>
053 * <p>
054 * See <a href="https://github.com/prometheus/pushgateway">https://github.com/prometheus/pushgateway</a>
055 */
056public class PushGateway {

061  private static final int MILLISECONDS_PER_SECOND = 1000;
062  /**
063   * Construct a Pushgateway, with the given address.
064   * <p>
065   * @param address  host:port or ip:port of the Pushgateway.
066   */
067  public PushGateway(String address) {
068    this(createURLSneakily("http://" + address));
069  }
070
071  /**
072   * Construct a Pushgateway, with the given URL.
073   * <p>
074   * @param serverBaseURL the base URL and optional context path of the Pushgateway server.
075   */
076  public PushGateway(URL serverBaseURL) {
077    this.gatewayBaseURL = URI.create(serverBaseURL.toString() + "/metrics/job/")
078      .normalize()
079      .toString();
080  }

082  /**
083   * Creates a URL instance from a String representation of a URL without throwing a checked exception.
084   * Required because you can't wrap a call to another constructor in a try statement.
085   *
086   * TODO: Remove this along with other deprecated methods before version 1.0 is released.
087   *
088   * @param urlString the String representation of the URL.
089   * @return The URL instance.
090   */
091  private static URL createURLSneakily(final String urlString) {
092    try {
093      return new URL(urlString);
094    } catch (MalformedURLException e) {
095      throw new RuntimeException(e);
096    }
097  }

268  void doRequest(CollectorRegistry registry, String job, Map<String, String> groupingKey, String method) throws IOException {
269    String url = gatewayBaseURL + URLEncoder.encode(job, "UTF-8");
270
271    if (groupingKey != null) {
272      for (Map.Entry<String, String> entry: groupingKey.entrySet()) {
273        url += "/" + entry.getKey() + "/" + URLEncoder.encode(entry.getValue(), "UTF-8");
274      }
275    }
276    HttpURLConnection connection = (HttpURLConnection) new URL(url).openConnection();
277    connection.setRequestProperty("Content-Type", TextFormat.CONTENT_TYPE_004);
278    if (!method.equals("DELETE")) {
279      connection.setDoOutput(true);
280    }
281    connection.setRequestMethod(method);
282
283    connection.setConnectTimeout(10 * MILLISECONDS_PER_SECOND);
284    connection.setReadTimeout(10 * MILLISECONDS_PER_SECOND);
285    connection.connect();
286
287    try {
288      if (!method.equals("DELETE")) {
289        BufferedWriter writer = new BufferedWriter(new OutputStreamWriter(connection.getOutputStream(), "UTF-8"));
290        TextFormat.write004(writer, registry.metricFamilySamples());
291        writer.flush();
292        writer.close();
293      }
294
295      int response = connection.getResponseCode();
296      if (response != HttpURLConnection.HTTP_ACCEPTED) {
297        throw new IOException("Response code from " + url + " was " + response);
298      }
299    } finally {
300      connection.disconnect();
301    }
302  }

304  /**
305   * Returns a grouping key with the instance label set to the machine's IP address.
306   * <p>
307   * This is a convenience function, and should only be used where you want to
308   * push per-instance metrics rather than cluster/job level metrics.
309  */
310  public static Map<String, String> instanceIPGroupingKey() throws UnknownHostException {
311    Map<String, String> groupingKey = new HashMap<String, String>();
312    groupingKey.put("instance", InetAddress.getLocalHost().getHostAddress());
313    return groupingKey;
314  }
